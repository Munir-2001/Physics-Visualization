import numpy as np
def SHM(amp,angfreq,phi,t):
    angfreq=np.radians(angfreq)
    phi=np.radians(phi)
    x=amp*np.cos((angfreq*t)+phi)
    v=-amp*angfreq*(np.sin((angfreq*t)+phi))  
    a=amp*angfreq*angfreq*(np.cos((angfreq*t)+phi))
    return(x,v,a)
